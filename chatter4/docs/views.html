def listener(io):
    r = redis.StrictRedis()
    r = r.pubsub()

    r.subscribe('chat')

    for m in r.listen():
        if not io.session.connected:
            return

        if m['type'] == 'message':
            data = loads(m['data'])
            io.send_event("chat", data)
